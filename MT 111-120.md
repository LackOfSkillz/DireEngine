from evennia import Command

class CmdWield(Command):
    key = "wield"

    def func(self):
        if not self.args:
            self.caller.msg("Wield what?")
            return

        obj = self.caller.search(self.args)
        if not obj:
            return

        self.caller.db.equipped_weapon = obj
        self.caller.msg(f"You wield {obj.key}.")
