      .

        logmsg "; $V0: $B4 $\"

        set t8 to "used"
        logmsg " $V0: $B4 $\"

: "$LOG_TRACKING"
! inputs:
!  np1  - player being impacted
!   v0  - skill impacted
!   v1  - new amount
!   s4  - rank or current or goal or field
!

    set a2 to (v3%60)
    set a3 to v3
    logmsg "  Tracking period is $A3 minutes ($A1 hr $A2 min).$\"
    set a4 to (a3-v2)
    logmsg "  The've rested a total of $A4 minutes during the tracking period.$\"
    set a4 to (v2*100/a3)
    logmsg "  $A4% of their time is being spent gaining new experience."
    if (v1>0) then
      logmsg "  $P1G$^ has gained $V1 exp during the current 2 minute cycle."
    else
      set v1 to (v1*-1)
      logmsg "  $P1G$^ has currently been resting for $V1 min."
    .
  .
!
!
  clear a0
  clear a1
  loop (a0 < 100)
    getskill a0 fieldexp of np1 for b4
    if (b4 > 0) then
      if (a1 = 0) then
        logmsg "  $P1G has the following field exp: $\"
        set a1 to 1
      .
      logmsg "$A0 = $B4;$\"
    .
    set a0 to (a0 + 1)
  .
  if (a1 = 0) then
    logmsg " $P1 has no field experience."
  else
    logmsg " done."
  .

!
  set a0 to np1:flag
  bitget a0 from 3 to 3 for a1
  if (a1 = 1) then
    logmsg "  $P1 is under the alternate experience system.$\"
  else
    logmsg "  $P1 is under the normal experience system.$\"
  .
!
!
  if (np1:sleeping <> 0) then
    logmsg "  $P1 is asleep."
  .
!
!
  logto np1 code 5
  msgp "$P1's experience tracking information has been logged."
.
