## Microtask 162 — Add condition helper (refined DR-style)
File modified: typeclasses/characters.py

Add:

def get_condition(self):
    hp, max_hp = self.get_hp()
    ratio = hp / max_hp if max_hp else 0

    if hp <= 0:
        return "dead"
    elif ratio < 0.1:
        return "near death"
    elif ratio < 0.25:
        return "badly wounded"
    elif ratio < 0.5:
        return "wounded"
    elif ratio < 0.75:
        return "bruised"
    else:
        return "in good shape"
