def normalize_stance(self):
    stance = self.db.stance or {"offense": 50, "defense": 50}

    total = stance["offense"] + stance["defense"]
    if total == 0:
        stance["offense"] = 50
        stance["defense"] = 50
        return

    stance["offense"] = int((stance["offense"] / total) * 100)
    stance["defense"] = int((stance["defense"] / total) * 100)

    self.db.stance = stance

            return
