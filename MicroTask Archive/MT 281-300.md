from typeclasses.wearables import Wearable

class Armor(Wearable):
    def at_object_creation(self):
        super().at_object_creation()

        self.db.item_type = "armor"
        self.db.armor_type = "light"  # light, chain, brigandine, plate

        # DR-style mitigation
        self.db.protection = 1        # flat reduction
        self.db.absorption = 0.1      # % reduction

        # hindrance
        self.db.maneuver_hindrance = 5
        self.db.stealth_hindrance = 5

        # coverage model
        self.db.covers = []  # ["chest", "arm", etc]

        # scaling system
        self.db.skill_scaling = {}

🧱 MT283 — Add get_armor_items()
def get_armor_items(self):
    equipment = self.get_equipment()
    armor = []

    for slot, item in equipment.items():
        if self.is_multi_slot(slot):
            for obj in item:
                if getattr(obj.db, "item_type", None) == "armor":
                    armor.append(obj)
        else:
            if item and getattr(item.db, "item_type", None) == "armor":
                armor.append(item)

    return armor

🧱 MT292 — Resolve armor scaling
def get_armor_effects(self, armor):
    effects = {}
    skill = ARMOR_SKILLS.get(armor.db.armor_type)
    rank = self.get_skill(skill)

    for tier in armor.db.skill_scaling.get(skill, []):
        if rank >= tier["rank"]:
            effects.update(tier["effects"])

    return effects
