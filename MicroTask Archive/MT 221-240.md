def can_hold_item(self, item):
    if not self.is_worn():
        return False, "You must be wearing it to use it."

    if len(self.db.contents) >= self.db.capacity:
        return False, "It cannot hold anything more."

    allowed = self.db.allowed_types
    if allowed and getattr(item.db, "item_type", None) not in allowed:

🧱 MT224 — Add store_item method

def store_item(self, item):
    can_hold, msg = self.can_hold_item(item)
    if not can_hold:
        return False, msg

    self.db.contents.append(item)
    item.location = None
    item.db.stored_in = self

    return True, f"You stow {item.key}."

def retrieve_item(self, item_name):
    for item in self.db.contents:
        if item.key.lower().startswith(item_name.lower()):
            self.db.contents.remove(item)
            item.location = self.db.worn_by
            item.db.stored_in = None
            return True, item, f"You draw {item.key}."

from evennia import Command

class CmdStow(Command):
    key = "stow"

    def func(self):
        if not self.args:
            self.caller.msg("Stow what?")
            return

        item = self.caller.search(self.args)
        if not item:
            return

        equipment = self.caller.get_equipment()

        for slot, obj in equipment.items():
            if isinstance(obj, list):
                for container in obj:
                    if getattr(container.db, "is_container", False):
                        success, msg = container.store_item(item)
                        if success:
                            self.caller.msg(msg)
                            return
            else:
                if getattr(obj, "is_container", False):
                    success, msg = obj.store_item(item)
                    if success:
                        self.caller.msg(msg)
                        return

        self.caller.msg("You have nowhere to stow that.")
