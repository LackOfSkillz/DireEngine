def get_equipment(self):
    equipment = self.db.equipment or {}

    # ensure all slots exist (safe migration)
    defaults = {
        "head": None,
        "face": None,
        "neck": None,
        "shoulders": None,
        "torso": None,
        "back": None,
        "arms": None,
        "hands": None,
        "waist": None,
        "legs": None,
        "feet": None,

        # multi-slot systems
        "fingers": [],

        # attachment systems (containers later)
        "belt_attach": [],
        "back_attach": [],
        "shoulder_attach": [],
    }

    updated = False
    for key, value in defaults.items():
        if key not in equipment:
            equipment[key] = value.copy() if isinstance(value, list) else value
            updated = True

    if updated:
        self.db.equipment = equipment

    return equipment
