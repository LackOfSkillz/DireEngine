🧠 Step 1 — Build Pathfinding (CLIENT OR SERVER)

Use BFS (simple and correct)

function findPath(startId, targetId, exits) {
    const queue = [[startId]];
    const visited = new Set();

    while (queue.length) {
        const path = queue.shift();
        const room = path[path.length - 1];

        if (room === targetId) return path;

        if (visited.has(room)) continue;
        visited.add(room);

        exits
            .filter(e => e.from === room)
            .forEach(e => {
                queue.push([...path, e.to]);
            });
    }

    return null;
}
