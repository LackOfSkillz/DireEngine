from evennia import Command

class CmdSkills(Command):
    key = "skills"

    def func(self):
        skills = self.caller.db.skills or {}
        if not skills:
            self.caller.msg("You know no skills.")
            return

        for name in skills:
            self.caller.msg(f"- {name}")
